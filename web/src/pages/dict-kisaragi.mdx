import { db } from "$src/server/db";
export const titles = (() => {
  const stmt = db.prepare(
    `SELECT DISTINCT title FROM heteronyms WHERE "from" = 'kisaragi_dict'`
  );
  stmt.pluck(true);
  return stmt.all();
})();
